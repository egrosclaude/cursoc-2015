4. Copiar y compilar el siguiente programa. Explicar el resultado.

main()
{
    double  x;
    int     i; 

    i = 1400;
    x = i;  /* conversión int/double */
    printf("x = %10.6le\ti = %d\n",x,i);
    x = 14.999;
    i = x;  /* conversión double/int */
    printf("x = %10.6le\ti = %d\n",x,i);
    x = 1.0e+60;
    i = x;
    printf("x = %10.6le\ti = %d\n",x,i);
}
